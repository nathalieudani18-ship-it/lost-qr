      contact: "Please call: 09061922690"
    },
    "1002": {
      name: "Juan Dela Cruz",
      contact: "Please call: 09123456789"
    }
  };

  if(items[id]){
    document.getElementById("owner").innerText =
      "This item belongs to " + items[id].name;
    document.getElementById("contact").innerText =
      items[id].contact;
  } else {
    document.getElementById("owner").innerText =
      "Item not registered.";
  }
